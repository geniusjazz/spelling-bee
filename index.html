  <button onclick="pronounceWord()">Pronounce</button>

  <input type="text" id="input" placeholder="Type the word">

  <button onclick="checkSpelling()">Check</button>

  <button onclick="nextWord()">Next</button>


  <script>

    const words = [

      { word: "apple", difficulty: "easy", definition: "a red or green fruit" },

      { word: "benevolent", difficulty: "hard", definition: "kind and generous" }

    ];

    let currentIndex = 0;


    function showWord() {

      const w = words[currentIndex];

      document.getElementById("difficulty").innerText = "Difficulty: " + w.difficulty;

      document.getElementById("definition").innerText = "Definition: " + w.definition;

      document.getElementById("input").value = "";

    }


    function pronounceWord() {

      const utterance = new SpeechSynthesisUtterance(words[currentIndex].word);

      utterance.lang = "en-US";

      speechSynthesis.speak(utterance);

    }


    function nextWord() {

      currentIndex = (currentIndex + 1) % words.length;

      showWord();

      pronounceWord();

    }


    function checkSpelling() {

      const userInput = document.getElementById("input").value.trim().toLowerCase();

      if (userInput === words[currentIndex].word) {

        alert("Correct!");

      } else {

        alert("Try again!");

      }

    }


    showWord();

    pronounceWord();

  </script>
